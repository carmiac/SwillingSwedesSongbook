\beginsong{Early One Morning}[by={English Traditional}]
\beginverse
Early one morning,
Just as the sun was rising,
I heard a maid sing,
In the valley below. 
\endverse
\beginchorus
Oh, don't deceive me,
Oh, never leave me,
How could you use
A poor maiden so? 
\endchorus
\beginverse
Remember the vows,
That you made to your Mary,
Remember the bower,
Where you vowed to be true, 
\endverse
\beginverse
Thus sang the poor maiden,
Her sorrows bewailing,
Thus sang the poor maid,
In the valley below. 
\endverse
\endsong
